package backgroundCheck;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.function.Function;
import java.util.stream.Collectors;

public interface BackgroundCheck {
  //Feedback: too much code here.
  
  //let's reduce to one function.
  
  //after that explore functional style

  @SafeVarargs
  public static Response evaluateCriteriaList(String ssn, Function<String, Response> ... criterion) {
    List<Response> responses= new ArrayList<>();
    Arrays.stream(criterion)
      .forEach(criteria -> responses.add(criteria.apply(ssn)));
    if (responses.stream().allMatch(response -> response.status)) return new Response(true, "");
    return new Response(false,
      responses.stream()
        .filter(response -> !response.status)
        .map(response -> response.reason)
        .collect(Collectors.joining(", ")));
  }
}
